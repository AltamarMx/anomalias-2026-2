import numpy as np
import matplotlib.pyplot as plt
import math
def gaussiana(x,mu=0, sigma=2):
    coeficiente = 1. / (sigma * np.sqrt(2 *np.pi))
    exponente    = -0.5 * ((x - mu) / sigma) ** 2
    gauss = coeficiente*np.exp(exponente)
    return gauss
